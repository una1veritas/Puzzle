CREATE TABLE IF NOT EXISTS kv (
  key   BLOB PRIMARY KEY,
  value BLOB
);
-- For lower overhead and smaller storage when you don't need the implicit rowid:
CREATE TABLE IF NOT EXISTS kv_no_rowid (
  key   BLOB PRIMARY KEY,
  value BLOB
) WITHOUT ROWID;
